CREATE TABLE DPDEPARTAMENTO
  (
    SIGLA_DEP VARCHAR(2) PRIMARY KEY,
    NOME_DEP  VARCHAR(30)
  );
DROP TABLE TB_CLIENTE;
DROP TABLE TB_VENDEDOR;
DROP TABLE TB_PRODUTO;
DROP TABLE TB_PEDIDO;
DROP TABLE TB_ITEM_PEDIDO;
CREATE TABLE TB_CLIENTE
  (
    CPF               VARCHAR(10) PRIMARY KEY NOT NULL,
    NOME              VARCHAR(30),
    ENDERECO          VARCHAR (30),
    CIDADE            VARCHAR(20),
    CEP               VARCHAR(8),
    UF                VARCHAR(2),
    TELEFONE_DDD      VARCHAR(3),
    TELEFONE_NUMERO   VARCHAR(8),
    LIMITE_CREDITO    NUMBER(12,2),
    LIMITE_DISPONIVEL NUMBER(12,2)
  ) ;
CREATE TABLE TB_VENDEDOR
  (
    CPF             VARCHAR(10) PRIMARY KEY NOT NULL,
    NOME            VARCHAR(30),
    ENDERECO        VARCHAR (30),
    CIDADE          VARCHAR(20),
    CEP             VARCHAR(8),
    UF              VARCHAR(2),
    TELEFONE_DDD    VARCHAR(3),
    TELEFONE_NUMERO VARCHAR(8),
    SALARIO_BASE    NUMBER (12,2),
    COMISSAO        NUMBER (12,2)
  );
CREATE TABLE TB_PRODUTO
  (
    CODIGO_PRODUTO  INTEGER PRIMARY KEY,
    DESCRICAO       VARCHAR(30),
    QTDE_DISPONIVEL INTEGER,
    PRECO_UNIT      NUMBER(12,2),
    ESTOQUE_MINIMO  INTEGER
  );
CREATE TABLE TB_PEDIDO
  (
    NUMERO INTEGER PRIMARY KEY,
    CPF_CLIENTE REFERENCES TB_CLIENTE,
    CPF_VENDEDOR REFERENCES TB_PRODUTO,
    STATUS      CHAR(1),
    DATA_PEDIDO VARCHAR(10),
    DATA_PAGTO  VARCHAR(10)
  );
  
CREATE TABLE TB_ITEM_PEDIDO
  (
    NUMERO_PEDIDO REFERENCES TB_PEDIDO,
    CODIGO_PRODUTO REFERENCES TB_PRODUTO,
    QT_VENDIDA INTEGER
  );
